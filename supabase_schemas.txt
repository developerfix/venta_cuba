create table public.device_tokens (
  id uuid not null default extensions.uuid_generate_v4 (),
  user_id text null,
  device_token text not null,
  platform text null default 'android'::text,
  is_active boolean null default true,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  constraint device_tokens_pkey primary key (id),
  constraint device_tokens_user_id_device_token_key unique (user_id, device_token)
) TABLESPACE pg_default;

create index IF not exists idx_device_tokens_platform on public.device_tokens using btree (platform) TABLESPACE pg_default;

create index IF not exists idx_device_tokens_updated on public.device_tokens using btree (updated_at desc) TABLESPACE pg_default;

create index IF not exists idx_device_tokens_is_active on public.device_tokens using btree (is_active) TABLESPACE pg_default;

create index IF not exists idx_device_tokens_user_id on public.device_tokens using btree (user_id) TABLESPACE pg_default;

create index IF not exists idx_device_tokens_active on public.device_tokens using btree (is_active) TABLESPACE pg_default;

create trigger update_device_tokens_updated_at BEFORE
update on device_tokens for EACH row
execute FUNCTION update_updated_at_column ();




create table public.messages (
  id uuid not null default extensions.uuid_generate_v4 (),
  chat_id text not null,
  message text null,
  send_by text not null,
  sender_name text null,
  time timestamp with time zone null default now(),
  image text null,
  message_type text null default 'text'::text,
  created_at timestamp with time zone null default now(),
  constraint messages_pkey primary key (id),
  constraint messages_chat_id_fkey foreign KEY (chat_id) references chats (id) on delete CASCADE
) TABLESPACE pg_default;

create index IF not exists idx_messages_chat_id on public.messages using btree (chat_id) TABLESPACE pg_default;

create index IF not exists idx_messages_time on public.messages using btree ("time" desc) TABLESPACE pg_default;


create table public.chats (
  id text not null,
  sender_id text not null,
  send_to_id text not null,
  sender_name text null,
  send_to_name text null,
  sender_image text null,
  send_to_image text null,
  message text null,
  time timestamp with time zone null default now(),
  send_by text null,
  user_device_token text null,
  send_to_device_token text null,
  is_messaged boolean null default false,
  sender_last_read_time timestamp with time zone null,
  recipient_last_read_time timestamp with time zone null,
  listing_id text null,
  listing_name text null,
  listing_image text null,
  listing_price text null,
  listing_location text null,
  message_type text null default 'text'::text,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  constraint chats_pkey primary key (id)
) TABLESPACE pg_default;

create index IF not exists idx_chats_sender_id on public.chats using btree (sender_id) TABLESPACE pg_default;

create index IF not exists idx_chats_send_to_id on public.chats using btree (send_to_id) TABLESPACE pg_default;

create index IF not exists idx_chats_time on public.chats using btree ("time" desc) TABLESPACE pg_default;






create table public.user_presence (
  user_id text not null,
  is_online boolean null default false,
  last_active_time timestamp with time zone null default now(),
  constraint user_presence_pkey primary key (user_id)
) TABLESPACE pg_default;
